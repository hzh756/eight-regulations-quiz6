        // 当前题目索引
        let currentQuestionIndex = 0;
        let selectedOption = -1;
        let answerSubmitted = false;
        let answeredQuestions = new Set(); // 记录已回答的题目

        // DOM元素
        const questionText = document.getElementById('question-text');
        const optionsContainer = document.getElementById('options-container');
        const answerSection = document.getElementById('answer-section');
        const submitBtn = document.getElementById('submit-btn');
        const prevBtn = document.getElementById('prev-btn');
        const nextBtn = document.getElementById('next-btn');
        const currentQuestionElement = document.getElementById('current-question');
        const questionCountElement = document.getElementById('question-count');
        const progressBar = document.getElementById('progress-bar');
        const questionsPanel = document.getElementById('questions-panel');
        const overlay = document.getElementById('overlay');
        const closePanel = document.getElementById('close-panel');
        const questionsGrid = document.getElementById('questions-grid');
        const searchInput = document.getElementById('search-input');
        const questionCard = document.getElementById('question-card');

        // 初始化 - 尝试从localStorage加载上次访问的题目
        function init() {
            const lastQuestionIndex = localStorage.getItem('lastQuestionIndex');
            if (lastQuestionIndex !== null) {
                currentQuestionIndex = parseInt(lastQuestionIndex);
            }
            
            // 加载已答题目
            const savedAnswered = localStorage.getItem('answeredQuestions');
            if (savedAnswered) {
                answeredQuestions = new Set(JSON.parse(savedAnswered));
            }
            
            renderQuestion(currentQuestionIndex);
        }

        // 渲染题目
        function renderQuestion(index) {
            const question = questions[index];
            currentQuestionElement.textContent = `题目 ${question.id}`;
            questionText.textContent = question.text;
            questionCountElement.textContent = `题目 ${index + 1}/${questions.length}`;
            
            // 更新进度条
            progressBar.style.width = `${((index + 1) / questions.length) * 100}%`;
            
            // 清空选项容器
            optionsContainer.innerHTML = '';
            
            // 添加选项
            question.options.forEach((option, i) => {
                const optionElement = document.createElement('div');
                optionElement.className = 'option';
                if (i === selectedOption) {
                    optionElement.classList.add('selected');
                }
                
                optionElement.innerHTML = `
                    <input type="radio" name="q${question.id}" id="q${question.id}o${i}">
                    <label for="q${question.id}o${i}" class="option-label">${option}</label>
                `;
                
                optionElement.addEventListener('click', () => {
                    if (answerSubmitted) return;
                    
                    // 移除之前的选择
                    document.querySelectorAll('.option').forEach(opt => {
                        opt.classList.remove('selected');
                    });
                    
                    // 设置当前选择
                    optionElement.classList.add('selected');
                    selectedOption = i;
                });
                
                optionsContainer.appendChild(optionElement);
            });
            
            // 重置答案区域
            answerSection.classList.remove('show');
            answerSection.innerHTML = `
                <div class="answer-title">
                    <i class="fas fa-check-circle"></i> 
                    正确答案：<span class="correct-answer">${question.options[question.correct]}</span>
                </div>
                <div class="explanation">
                    ${question.explanation}
                </div>
            `;
            
            // 重置状态
            answerSubmitted = false;
            selectedOption = -1;
            
            // 更新按钮状态
            prevBtn.disabled = index === 0;
            nextBtn.disabled = index === questions.length - 1;
            
            // 如果此题已答过，显示答案
            if (answeredQuestions.has(question.id)) {
                showAnswer();
            }
            
            // 根据题目ID设置主题
            if (question.id >= 101) {
                questionCard.classList.add('red-theme');
            } else {
                questionCard.classList.remove('red-theme');
            }
        }

        // 显示答案
        function showAnswer() {
            answerSection.classList.add('show');
            
            // 标记正确和错误的选项
            const options = document.querySelectorAll('.option');
            options.forEach((option, i) => {
                if (i === selectedOption) {
                    option.classList.add(selectedOption === questions[currentQuestionIndex].correct ? 'correct' : 'incorrect');
                } else if (i === questions[currentQuestionIndex].correct) {
                    option.classList.add('correct');
                }
            });
            
            answerSubmitted = true;
        }

        // 提交答案
        submitBtn.addEventListener('click', () => {
            if (selectedOption === -1) {
                alert('请选择一个答案！');
                return;
            }
            
            if (answerSubmitted) return;
            
            showAnswer();
            
            // 记录已回答的题目
            answeredQuestions.add(questions[currentQuestionIndex].id);
            localStorage.setItem('answeredQuestions', JSON.stringify(Array.from(answeredQuestions)));
        });

        // 上一题
        prevBtn.addEventListener('click', () => {
            if (currentQuestionIndex > 0) {
                currentQuestionIndex--;
                renderQuestion(currentQuestionIndex);
                saveCurrentPosition();
            }
        });

        // 下一题
        nextBtn.addEventListener('click', () => {
            if (currentQuestionIndex < questions.length - 1) {
                currentQuestionIndex++;
                renderQuestion(currentQuestionIndex);
                saveCurrentPosition();
            }
        });

        // 保存当前位置
        function saveCurrentPosition() {
            localStorage.setItem('lastQuestionIndex', currentQuestionIndex);
        }

        // 显示题目导航面板
        searchInput.addEventListener('click', () => {
            overlay.classList.add('show');
            questionsPanel.classList.add('show');
            renderQuestionsGrid();
        });

        // 关闭题目导航面板
        closePanel.addEventListener('click', closeQuestionsPanel);
        overlay.addEventListener('click', closeQuestionsPanel);
        
        function closeQuestionsPanel() {
            overlay.classList.remove('show');
            questionsPanel.classList.remove('show');
        }

        // 渲染题目导航网格
        function renderQuestionsGrid() {
            questionsGrid.innerHTML = '';
            
            questions.forEach((question, index) => {
                const questionItem = document.createElement('div');
                questionItem.className = 'question-item';
                questionItem.textContent = question.id;
                
                // 标记当前题目
                if (index === currentQuestionIndex) {
                    questionItem.classList.add('current');
                }
                
                // 标记已答题目
                if (answeredQuestions.has(question.id)) {
                    questionItem.classList.add('answered');
                }
                
                questionItem.addEventListener('click', () => {
                    currentQuestionIndex = index;
                    renderQuestion(currentQuestionIndex);
                    saveCurrentPosition();
                    closeQuestionsPanel();
                });
                
                questionsGrid.appendChild(questionItem);
            });
        }

        // 搜索题目
        searchInput.addEventListener('input', () => {
            const searchTerm = searchInput.value.toLowerCase().trim();
            if (searchTerm.length === 0) return;
            
            // 尝试按题号搜索
            if (/^\d+$/.test(searchTerm)) {
                const questionId = parseInt(searchTerm);
                const foundIndex = questions.findIndex(q => q.id === questionId);
                if (foundIndex !== -1) {
                    currentQuestionIndex = foundIndex;
                    renderQuestion(currentQuestionIndex);
                    saveCurrentPosition();
                    return;
                }
            }
            
            // 按内容搜索
            const foundIndex = questions.findIndex(q => 
                q.text.toLowerCase().includes(searchTerm) || 
                q.explanation.toLowerCase().includes(searchTerm)
            );
            
            if (foundIndex !== -1) {
                currentQuestionIndex = foundIndex;
                renderQuestion(currentQuestionIndex);
                saveCurrentPosition();
            }
        });

        // 页面卸载时保存当前题目位置
        window.addEventListener('beforeunload', saveCurrentPosition);
        
        // 初始化页面
        init();
    </script>
